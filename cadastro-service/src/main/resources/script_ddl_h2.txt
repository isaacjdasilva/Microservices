create table tb_pessoa (
   id bigint generated by default as identity,
    nome varchar(255),
    ic_pessoa char(1) not null,
    primary key (id)
);

create table tb_pessoa_fisica (
   cpf varchar(255),
    id bigint not null,
    primary key (id)
); 

create table tb_pessoa_juridica (
   cnpj varchar(255),
    id bigint not null,
    primary key (id)
);

alter table tb_pessoa_fisica 
   add constraint fk_tb_pessoa_fisica_pessoa 
   foreign key (id) 
   references tb_pessoa;

alter table tb_pessoa_juridica 
   add constraint fk_tb_pessoa_juridica_pessoa 
   foreign key (id) 
   references tb_pessoa;
